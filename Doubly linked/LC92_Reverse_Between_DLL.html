def reverseBetween(head, left, right):
dummy = DLLNode(0); dummy.next=head; head.prev=dummy
prev_anchor = dummy
for _ in range(left-1): prev_anchor=prev_anchor.next
left_node = prev_anchor.next
right_node = left_node
for _ in range(right-left): right_node=right_node.next
next_anchor = right_node.next
# Swap next/prev for every node in segment
curr = left_node
while curr != next_anchor:
curr.next, curr.prev = curr.prev, curr.next
curr = curr.prev # move forward (was .next)
# Reconnect segment to surrounding list
prev_anchor.next = right_node
right_node.prev = prev_anchor
left_node.next = next_anchor
if next_anchor: next_anchor.prev = left_node
return dummy.next
