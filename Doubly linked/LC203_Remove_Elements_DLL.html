def removeElements(head, val):
dummy = DLLNode(0)
dummy.next = head; head.prev = dummy
curr = dummy
while curr.next:
if curr.next.val == val:
to_remove = curr.next
curr.next = to_remove.next
if to_remove.next:
to_remove.next.prev = curr
to_remove.next = to_remove.prev = None
else:
curr = curr.next
if dummy.next: dummy.next.prev = None
return dummy.next
