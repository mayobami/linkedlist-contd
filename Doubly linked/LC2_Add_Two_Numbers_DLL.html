def addTwoNumbers(l1, l2):
# Find tails (least significant digits)
tail1 = l1
while tail1.next: tail1=tail1.next
tail2 = l2
while tail2.next: tail2=tail2.next
p1, p2 = tail1, tail2
carry = 0; result_head = None
while p1 or p2 or carry:
total = carry
if p1: total+=p1.val; p1=p1.prev
if p2: total+=p2.val; p2=p2.prev
carry, digit = divmod(total, 10)
# Prepend to result
new_node = DLLNode(digit)
new_node.next = result_head
if result_head: result_head.prev = new_node
result_head = new_node
return result_head
