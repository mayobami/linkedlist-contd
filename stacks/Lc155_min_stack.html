LC 155 — Min Stack

Easy All ops O(1) Space O(n)
class MinStack: def __init__(self): self.stack = [] self.min_stack = [] def push(self, val: int): self.stack.append(val) cur_min = val if not self.min_stack \ else min(val, self.min_stack[-1]) self.min_stack.append(cur_min) def pop(self): self.stack.pop() self.min_stack.pop() def top(self) -> int: return self.stack[-1] def getMin(self) -> int: return self.min_stack[-1]
