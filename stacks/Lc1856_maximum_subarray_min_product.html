LC 1856 — Maximum Subarray Min-Product

Hard Time O(n) Space O(n)
def maxSumMinProduct(self, nums): MOD = 10**9 + 7 n = len(nums) prefix = [0] * (n + 1) for i in range(n): prefix[i+1] = prefix[i] + nums[i] stack, ans = [], 0 for i in range(n): while stack and nums[stack[-1]] > nums[i]: idx = stack.pop() left = stack[-1] + 1 if stack else 0 s = prefix[i] - prefix[left] ans = max(ans, nums[idx] * s) stack.append(i) while stack: # cleanup idx = stack.pop() left = stack[-1] + 1 if stack else 0 s = prefix[n] - prefix[left] ans = max(ans, nums[idx] * s) return ans % MOD
