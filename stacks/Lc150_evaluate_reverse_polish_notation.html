LC 150 — Evaluate Reverse Polish Notation

Medium Time O(n) Space O(n)
def evalRPN(self, tokens): stack = [] for tok in tokens: if tok in '+-*/': b = stack.pop() # top (pushed last) a = stack.pop() # below (pushed first) if tok == '+': stack.append(a + b) elif tok == '-': stack.append(a - b) elif tok == '*': stack.append(a * b) else: stack.append(int(a / b)) else: stack.append(int(tok)) return stack[0]
