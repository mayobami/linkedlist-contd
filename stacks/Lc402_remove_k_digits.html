LC 402 — Remove K Digits

Medium Time O(n) Space O(n)
def removeKdigits(self, num, k): stack = [] for d in num: while k and stack and stack[-1] > d: stack.pop() k -= 1 stack.append(d) stack = stack[:-k] if k else stack # trim if k remains return ''.join(stack).lstrip('0') or '0'
