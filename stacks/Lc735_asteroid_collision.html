LC 735 — Asteroid Collision

Medium Time O(n) Space O(n)
def asteroidCollision(self, asteroids): stack = [] for a in asteroids: alive = True while alive and stack and a < 0 < stack[-1]: if stack[-1] < -a: # top smaller → top destroyed stack.pop() elif stack[-1] == -a: # equal → both destroyed stack.pop(); alive = False else: # top larger → current destroyed alive = False if alive: stack.append(a) return stack
