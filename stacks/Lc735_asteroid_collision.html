LC 735 — Asteroid Collision

Medium Time O(n) Space O(n)
def asteroidCollision(self, asteroids): stack = [] for a in asteroids: while stack and a < 0 and stack[-1] > 0: diff = a + stack[-1] if diff < 0: # top smaller → top destroyed stack.pop() elif diff > 0: # current smaller → current destroyed a = 0 else: # equal → both destroyed a = 0 stack.pop() if a: stack.append(a) return stack
