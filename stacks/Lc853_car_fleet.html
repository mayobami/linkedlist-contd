LC 853 — Car Fleet

Medium Time O(n log n) Space O(n)
def carFleet(self, target, position, speed): pairs = sorted(zip(position, speed), reverse=True) stack = [] # fleet finish times for pos, spd in pairs: t = (target - pos) / spd if not stack or t > stack[-1]: stack.append(t) return len(stack)
